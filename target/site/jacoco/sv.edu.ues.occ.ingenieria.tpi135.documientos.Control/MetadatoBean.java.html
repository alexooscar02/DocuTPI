package sv.edu.ues.occ.ingenieria.tpi135.documientos.Control;

import jakarta.ejb.LocalBean;
import jakarta.ejb.Stateless;
import jakarta.persistence.EntityManager;
import jakarta.persistence.PersistenceContext;
import java.io.Serializable;
import sv.edu.ues.occ.ingenieria.tpi135.documientos.entity.Metadato;

/**
 *
 * @author alexo
 */
@Stateless
@LocalBean
public class MetadatoBean extends AbstractDataAccess<Metadato> implements Serializable {

    @PersistenceContext(unitName = "documientos-PU")
    EntityManager em;

    public MetadatoBean() {
        super(Metadato.class);
    }

    @Override
    public EntityManager getEntityManager() {
        return em;
    }

        public boolean verificarMetadatoPerteneceTaxonomia(Long idDocumento, Long idMetadata) {
        Long count = (Long) em.createNamedQuery("Metadato.findCountByDocumentoAndMetadata")
                .setParameter("idDocumento", idDocumento)
                .setParameter("idMetadata", idMetadata)
                .getSingleResult();

        return count > 0;
    }

}
